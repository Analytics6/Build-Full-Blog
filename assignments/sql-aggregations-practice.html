Example

SQL
-- Total salary per department
SELECT d.department_name, SUM(e.salary) AS total_salary
FROM hr.employees e
INNER JOIN hr.departments d ON e.department_id = d.department_id
GROUP BY d.department_name
ORDER BY total_spent DESC;
